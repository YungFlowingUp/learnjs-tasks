function pow(x,n) // нет пробела после запятой
{ //следует ставить фигурную скобку на строке с объявлением функции
  let result=1; // нет пробелов вокруг оператора присваивания
  for(let i=0;i<n;i++) {result*=x;} // нет пробела после for, нет пробелов внутри скобок, фигурные скобки не нужны, а лучше разделить на несколько строк
  return result;
}
